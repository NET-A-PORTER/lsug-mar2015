class PriceClient {
  
  val pipeline = (
    logRequest(log)
    ~> addHeader(`Accept-Encoding`(gzip))
    ~> sendReceive
    ~> decode(Gzip)
    ~> logResponse(log)
    ~> unmarshal[Price]
  )

  val resp = pipeline(HttpRequest(...))
}

spray-http Testing

class PriceClient(transport: ActorRef) {
  
  val pipeline = (
    logRequest(log)
    ~> addHeader(`Accept-Encoding`(gzip))
    ~> sendReceive(transport)
    ~> decode(Gzip)
    ~> logResponse(log)
    ~> unmarshal[Price]
  )

  val resp = pipeline(HttpRequest(...))
}
